<script>
    // ----------------------------------------------------------------
    // 1. DONNÉES SIMULÉES
    // ----------------------------------------------------------------
    const pairs = [
        { pair: 'BTC/USD', price: 64250, change: 1.8 },
        { pair: 'ETH/USD', price: 3450, change: -0.7 },
        { pair: 'SOL/USD', price: 172.5, change: 4.2 },
        { pair: 'XRP/USD', price: 0.612, change: 0.3 }
    ];

    // Ordres simulés
    const orders = [
        { pair: 'BTC/USD', type: 'Achat', price: 64100, volume: 0.25, status: 'Exécuté' },
        { pair: 'ETH/USD', type: 'Vente', price: 3480, volume: 1.5, status: 'En attente' },
        { pair: 'SOL/USD', type: 'Achat', price: 170.0, volume: 12, status: 'Exécuté' },
        { pair: 'BTC/USD', type: 'Vente', price: 64500, volume: 0.1, status: 'Annulé' }
    ];

    // Signal actuel
    let currentSignal = 'NEUTRE';
    let currentConfidence = 72;
    let currentRsi = 54;

    // ----------------------------------------------------------------
    // 2. AFFICHAGE PRIX
    // ----------------------------------------------------------------
    function renderPrices() {
        const grid = document.getElementById('priceGrid');
        grid.innerHTML = '';
        pairs.forEach(p => {
            const card = document.createElement('div');
            card.className = 'price-card';
            const changeClass = p.change >= 0 ? 'positive' : 'negative';
            card.innerHTML = `
                <div class="pair">${p.pair}</div>
                <div class="price">$${p.price.toLocaleString()}</div>
                <div class="change ${changeClass}">${p.change >= 0 ? '+' : ''}${p.change}%</div>
            `;
            grid.appendChild(card);
        });
    }

    // ----------------------------------------------------------------
    // 3. TABLEAU ORDRES
    // ----------------------------------------------------------------
    function renderOrders() {
        const tbody = document.getElementById('ordersBody');
        tbody.innerHTML = '';
        orders.forEach(o => {
            const tr = document.createElement('tr');
            const badgeClass = o.type === 'Achat' ? 'buy-badge' : 'sell-badge';
            tr.innerHTML = `
                <td>${o.pair}</td>
                <td><span class="badge ${badgeClass}">${o.type}</span></td>
                <td>$${o.price.toLocaleString()}</td>
                <td>${o.volume}</td>
                <td>${o.status}</td>
            `;
            tbody.appendChild(tr);
        });
    }

    // ----------------------------------------------------------------
    // 4. GRAPHIQUE BTC (Chart.js)
    // ----------------------------------------------------------------
    let chartInstance = null;

    function initChart() {
        const ctx = document.getElementById('btcChart').getContext('2d');
        const labels = Array.from({ length: 24 }, (_, i) => `${i}h`);
        const data = [63200, 63500, 63800, 64000, 64200, 64500, 64300, 64100, 63900, 63800, 64000, 64250, 64400, 64600, 64800, 64700, 64500, 64300, 64200, 64300, 64400, 64550, 64400, 64250];

        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: 'BTC/USD',
                    data: data,
                    borderColor: '#f7b731',
                    backgroundColor: 'rgba(247, 183, 49, 0.1)',
                    borderWidth: 3,
                    pointRadius: 2,
                    pointBackgroundColor: '#f7b731',
                    tension: 0.3,
                    fill: true
                }]
            },
            options: {
                responsive: true,
                plugins: {
                    legend: { display: false },
                    tooltip: { enabled: true }
                },
                scales: {
                    x: { grid: { color: '#1e2636' }, ticks: { color: '#8a9bb5' } },
                    y: { grid: { color: '#1e2636' }, ticks: { color: '#8a9bb5' } }
                }
            }
        });
    }

    // ----------------------------------------------------------------
    // 5. SIGNAL (simulation aléatoire)
    // ----------------------------------------------------------------
    function updateSignal() {
        const rnd = Math.random();
        let signal, confidence, rsi;
        if (rnd < 0.35) {
            signal = 'ACHAT';
            confidence = Math.floor(65 + Math.random() * 25);
            rsi = Math.floor(30 + Math.random() * 20);
        } else if (rnd < 0.70) {
            signal = 'VENTE';
            confidence = Math.floor(60 + Math.random() * 25);
            rsi = Math.floor(60 + Math.random() * 20);
        } else {
            signal = 'NEUTRE';
            confidence = Math.floor(40 + Math.random() * 30);
            rsi = Math.floor(40 + Math.random() * 20);
        }

        currentSignal = signal;
        currentConfidence = confidence;
        currentRsi = rsi;

        const signalEl = document.getElementById('signalValue');
        signalEl.textContent = signal;
        signalEl.className = 'signal-value ' + signal.toLowerCase();

        document.getElementById('confidence').textContent = confidence + '%';
        document.getElementById('rsiValue').textContent = rsi;
    }

    // ----------------------------------------------------------------
    // 6. RAFRAÎCHISSEMENT GLOBAL
    // ----------------------------------------------------------------
    function refreshData() {
        // Simule une mise à jour des prix
        pairs.forEach(p => {
            const variation = (Math.random() - 0.5) * 80;
            p.price = Math.round((p.price + variation) * 100) / 100;
            p.change = +(p.change + (Math.random() - 0.5) * 1.2).toFixed(1);
        });
        renderPrices();
        updateSignal();
        // Met à jour le graphique avec une nouvelle valeur
        if (chartInstance) {
            const newData = chartInstance.data.datasets[0].data;
            newData.shift();
            newData.push(Math.round((64250 + (Math.random() - 0.5) * 400) * 100) / 100);
            chartInstance.update();
        }
    }

    // ----------------------------------------------------------------
    // 7. INIT
    // ----------------------------------------------------------------
    renderPrices();
    renderOrders();
    initChart();
    updateSignal();

    // Auto-refresh toutes les 8 secondes
    setInterval(refreshData, 8000);
</script>
